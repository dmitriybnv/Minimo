key = 0;

snakeColumn = 0;
snakeData = 1;

GPUSelect(snakeColumn)
GPUAdd(snakeData)

func loop
    keyboardRead(key)

    if (key == 1)
        if (snakeColumn > 0)
            GPUSelect(snakeColumn)
            GPUSubtract(snakeData)

            snakeColumn = snakeColumn - 1;

            GPUSelect(snakeColumn)
            GPUAdd(snakeData)
        endif
    endif

    if (key == 3)
        if (snakeColumn < 19)
            GPUSelect(snakeColumn)
            GPUSubtract(snakeData)

            snakeColumn = snakeColumn + 1;

            GPUSelect(snakeColumn)
            GPUAdd(snakeData)
        endif
    endif

    if (key == 4)
        if (snakeData < 128)
            GPUSelect(snakeColumn)
            GPUSubtract(snakeData)

            shiftLeft(snakeData)

            GPUSelect(snakeColumn)
            GPUAdd(snakeData)
        endif
    endif

    if (key == 2)
        if (snakeData > 1)
            GPUSelect(snakeColumn)
            GPUSubtract(snakeData)

            shiftRight(snakeData)

            GPUSelect(snakeColumn)
            GPUAdd(snakeData)
        endif
    endif

    keyboardClear()

    loop()
endfunc

loop()
