func drawSnakeOnce
    index = 0;

    index2 = 0;

    #drawSnakeOnceLoop#
        snakeX = baseAddress + index2;

        temp = 0;

        RAMSelect(snakeX)
        RAMRead(temp)

        GPUSelect(temp)

        snakeY = baseAddress + index2 + 1;

        RAMSelect(snakeY)
        RAMRead(temp)

        GPUAdd(temp)

        index = index + 1;

        index2 = index2 + 2;

        if (index < snakeLength)
           goto drawSnakeOnceLoop$
        endif
endfunc

func handleKeypress
    firstSnakeX = 0;

    firstSnakeY = 0;

    tempFirstSnakeX = 0;

    tempFirstSnakeY = 0;

    RAMSelect(baseAddress)
    RAMRead(firstSnakeX)
    RAMRead(tempFirstSnakeX)

    tempAddress = baseAddress + 1;

    RAMSelect(tempAddress)
    RAMRead(firstSnakeY)
    RAMRead(tempFirstSnakeY)

    changeFlag = 0;

    if (key == 1)
        if (firstSnakeX > 0)
            changeFlag = 1;

            tempFirstSnakeX = tempFirstSnakeX - 1;
        endif
    endif

    if (key == 3)
        if (firstSnakeX < 19)
            changeFlag = 1;

            tempFirstSnakeX = tempFirstSnakeX + 1;
        endif
    endif

    if (key == 4)
        if (firstSnakeY < 128)
            changeFlag = 1;

            shiftLeft(tempFirstSnakeY)
        endif
    endif

    if (key == 2)
        if (firstSnakeY > 1)
            changeFlag = 1;

            shiftRight(tempFirstSnakeY)
        endif
    endif

    if (changeFlag == 1)
        GPUSelect(firstSnakeX)
        GPUSubtract(firstSnakeY)

        GPUSelect(tempFirstSnakeX)
        GPUAdd(tempFirstSnakeY)

        previousX = firstSnakeX;
        previousY = firstSnakeY;

        RAMSelect(baseAddress)
        RAMWrite(tempFirstSnakeX)

        RAMSelect(tempAddress)
        RAMWrite(tempFirstSnakeY)

        if (snakeLength > 1)
            index = 1;

            index2 = 2;

            #handleKeypressLoop#
                temp = 0;

                snakeX = baseAddress + index2;

                RAMSelect(snakeX)
                RAMRead(temp)

                GPUSelect(temp)

                snakeY = baseAddress + index2 + 1;

                RAMSelect(snakeY)
                RAMRead(temp)

                GPUSubtract(temp)

                index = index + 1;

                index2 = index2 + 2;

                if (index < snakeLength)
                   goto handleKeypressLoop$
                endif
        endif
    endif
endfunc
